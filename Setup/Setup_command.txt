# Clone repository
git clone https://github.com/Kurubagagana/AI-Powered-SIEM-Alert-Investigation-Chatbot.git

# Move into project directory
cd AI-Powered-SIEM-Alert-Investigation-Chatbot

# Check project files
ls

# Check Python version
python --version

# Create virtual environment
python -m venv venv

# Activate virtual environment (Windows)
venv\Scripts\activate

# Activate virtual environment (Linux/Mac)
source venv/bin/activate

# Install project dependencies
pip install -r requirements.txt

# Check installed Python packages
pip list

# Create environment configuration file
touch .env

# Run Flask SIEM application
python app.py

# Run application in debug mode (optional)
python -m flask run

# Check running Flask server
http://127.0.0.1:5000
